## Quick Start

Get started with the SuperMemory SDK in just a few lines:

```typescript
import { SuperMemory } from '@supermemory/sdk';

const supermemory = new SuperMemory({
  apiKey: 'your_api_key_here',
  baseUrl: 'https://api.vevibe.com', // Optional, defaults to production
});

// Add a memory
const memory = await supermemory.memories.create({
  type: 'page',
  content: 'https://example.com/article',
  spaces: ['work-space-id']
});

// Search memories
const results = await supermemory.search({
  query: 'machine learning best practices',
  limit: 10
});

// Chat with your knowledge
const chatResponse = await supermemory.chat({
  messages: [
    { role: 'user', content: 'What have I learned about React hooks?' }
  ]
});
```

## Configuration

### Environment Variables

Set up your environment variables:

```bash
# .env
SUPERMEMORY_API_KEY=your_api_key_here
SUPERMEMORY_BASE_URL=https://api.vevibe.com  # Optional
```

### SDK Configuration Options

```typescript
interface SuperMemoryConfig {
  apiKey: string;
  baseUrl?: string;          // Default: 'https://api.vevibe.com'
  timeout?: number;          // Default: 30000 (30 seconds)
  retries?: number;          // Default: 3
  retryDelay?: number;       // Default: 1000 (1 second)
  userAgent?: string;        // Custom user agent
}

const supermemory = new SuperMemory({
  apiKey: process.env.SUPERMEMORY_API_KEY!,
  timeout: 60000,  // 60 seconds
  retries: 5,
  retryDelay: 2000,
});
```

## Error Handling

### Built-in Error Types

```typescript
import { SuperMemoryError, RateLimitError, AuthenticationError } from '@supermemory/sdk';

try {
  const memory = await supermemory.memories.create({
    type: 'page',
    content: 'https://example.com'
  });
} catch (error) {
  if (error instanceof AuthenticationError) {
    console.error('Invalid API key:', error.message);
  } else if (error instanceof RateLimitError) {
    console.error('Rate limit exceeded. Retry after:', error.retryAfter);
  } else if (error instanceof SuperMemoryError) {
    console.error('API error:', error.message, error.details);
  } else {
    console.error('Unexpected error:', error);
  }
}
```

### Automatic Retry Logic

```typescript
// SDK automatically retries failed requests
const supermemory = new SuperMemory({
  apiKey: 'your_api_key',
  retries: 3,
  retryDelay: 1000,
  // Exponential backoff: 1s, 2s, 4s
  retryDelayMultiplier: 2
});

// Custom retry logic for specific operations
const memoryWithRetry = await supermemory.memories.create(
  {
    type: 'page',
    content: 'https://example.com'
  },
  {
    retries: 5,
    retryDelay: 2000
  }
);
```

// All responses are properly typed
const memory: Memory = await supermemory.memories.get('uuid');
const spaces: Space[] = await supermemory.spaces.list();
const results: SearchResult[] = (await supermemory.search({ query: 'test' })).results;

// Function parameters are type-checked

  type: 'page', // Type: 'page' | 'note' | 'document' | 'tweet'

  spaces: ['uuid'] // Type: string[]
});
```

```

### Middleware and Interceptors

```typescript
// Add request/response interceptors
supermemory.addRequestInterceptor((config) => {
  console.log('Making request to:', config.url);
  return config;
});

supermemory.addResponseInterceptor(
  (response) => {
    console.log('Response received:', response.status);
    return response;
  },
  (error) => {
    console.error('Request failed:', error.message);
    return Promise.reject(error);
  }
);
```

### Batch Operations with Progress Tracking

```typescript
async function bulkImportWithProgress(urls: string[]) {
  const response = await supermemory.memories.bulkAdd({
    urls,
    spaces: ['import-space-uuid']
  });

  let processed = 0;
  let succeeded = 0;
  let failed = 0;

  for await (const update of response) {
    processed++;
    
    if (update.success) {
      succeeded++;
      console.log(`✓ Imported: ${update.url}`);
    } else {
      failed++;
      console.log(`✗ Failed: ${update.url} - ${update.error}`);
    }

    const progress = (processed / urls.length) * 100;
    console.log(`Progress: ${progress.toFixed(1)}% (${succeeded} succeeded, ${failed} failed)`);
  }

  console.log(`\nImport complete: ${succeeded}/${urls.length} successful`);
}
```

## Integration Examples

### React Hook

```typescript
// useSupermemory.ts
import { useState, useEffect } from 'react';
import { SuperMemory, Memory } from '@supermemory/sdk';

const supermemory = new SuperMemory({
  apiKey: process.env.REACT_APP_SUPERMEMORY_API_KEY!
});

export function useMemories(spaceId?: string) {
  const [memories, setMemories] = useState<Memory[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function fetchMemories() {
      try {
        setLoading(true);
        const response = await supermemory.memories.list({
          spaceId,
          count: 50
        });
        setMemories(response.items);
      } catch (err) {
        setError(err instanceof Error ? err.message : 'Unknown error');
      } finally {
        setLoading(false);
      }
    }

    fetchMemories();
  }, [spaceId]);

  return { memories, loading, error };
}

// Component usage
function MemoryList({ spaceId }: { spaceId?: string }) {
  const { memories, loading, error } = useMemories(spaceId);

  if (loading) return <div>Loading memories...</div>;
  if (error) return <div>Error: {error}</div>;

  return (
    <div>
      {memories.map(memory => (
        <div key={memory.uuid}>
          <h3>{memory.title}</h3>
          <p>{memory.description}</p>
        </div>
      ))}
    </div>
  );
}
```

### Express.js API

```typescript
import express from 'express';
import { SuperMemory } from '@supermemory/sdk';

const app = express();
const supermemory = new SuperMemory({
  apiKey: process.env.SUPERMEMORY_API_KEY!
});

app.use(express.json());

// Proxy search endpoint
app.post('/api/search', async (req, res) => {
  try {
    const { query, limit = 10 } = req.body;
    const results = await supermemory.search({ query, limit });
    res.json(results);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Add memory endpoint
app.post('/api/memories', async (req, res) => {
  try {
    const memory = await supermemory.memories.create(req.body);
    res.status(201).json(memory);
  } catch (error) {
    res.status(400).json({ error: error.message });
  }
});

app.listen(3000, () => {
  console.log('Server running on port 3000');
});
```

## Testing

### Unit Testing with Jest

```typescript
// supermemory.test.ts
import { SuperMemory } from '@supermemory/sdk';

// Mock the SDK for testing
jest.mock('@supermemory/sdk');

describe('SuperMemory Integration', () => {
  let supermemory: SuperMemory;

  beforeEach(() => {
    supermemory = new SuperMemory({
      apiKey: 'test-api-key'
    });
  });

  test('should create memory', async () => {
    const mockMemory = {
      uuid: 'test-uuid',
      title: 'Test Memory',
      type: 'note',
      content: 'Test content'
    };

    (supermemory.memories.create as jest.Mock).mockResolvedValue(mockMemory);

    const memory = await supermemory.memories.create({
      type: 'note',
      content: 'Test content'
    });

    expect(memory).toEqual(mockMemory);
    expect(supermemory.memories.create).toHaveBeenCalledWith({
      type: 'note',
      content: 'Test content'
    });
  });
});
```
