## Overview

SuperMemory API uses Bearer token authentication with API keys. This approach is simple, secure, and widely supported across programming languages and tools.

## API Key Authentication

### Getting Your API Key

<Steps>
  <Step title="Access Dashboard">
    Log in to your [SuperMemory dashboard](https://supermemory.ai)
  </Step>
  <Step title="Navigate to Settings">
    Go to Settings → Integrations or Settings → API Keys
  </Step>
  <Step title="Generate Key">
    Click "Generate API Key" to create a new key
  </Step>
  <Step title="Copy Securely">
    Copy the key immediately - it's only shown once!
  </Step>
</Steps>

<Warning>
API keys are only displayed once when generated. Make sure to copy and store your key securely before leaving the page.
</Warning>

### Using Your API Key

Include your API key in the `Authorization` header using the Bearer token format:

```http
Authorization: Bearer your_api_key_here
```

## Authentication Examples

### cURL

```bash
# Basic authentication example
curl -H "Authorization: Bearer YOUR_API_KEY" \
     https://api.vevibe.com/v1/user

# With request body
curl -X POST https://api.vevibe.com/v1/add \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "type": "page",
    "content": "https://example.com/article"
  }'
```

### JavaScript/Node.js

```javascript
const API_KEY = 'your_api_key_here';
const BASE_URL = 'https://api.vevibe.com';

// Using fetch
async function apiRequest(endpoint, options = {}) {
  const response = await fetch(`${BASE_URL}${endpoint}`, {
    ...options,
    headers: {
      'Authorization': `Bearer ${API_KEY}`,
      'Content-Type': 'application/json',
      ...options.headers,
    },
  });
  
  if (!response.ok) {
    throw new Error(`API request failed: ${response.statusText}`);
  }
  
  return response.json();
}

// Usage examples
const user = await apiRequest('/v1/user');

const memory = await apiRequest('/v1/add', {
  method: 'POST',
  body: JSON.stringify({
    type: 'note',
    content: 'This is a test note'
  }),
});
```

### Python

### Go

```go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
    "os"
)

const baseURL = "https://api.vevibe.com"

func makeRequest(method, endpoint string, body interface{}) (*http.Response, error) {
    var reqBody []byte
    var err error
    
    if body != nil {
        reqBody, err = json.Marshal(body)
        if err != nil {
            return nil, err
        }
    }
    
    req, err := http.NewRequest(method, baseURL+endpoint, bytes.NewBuffer(reqBody))
    if err != nil {
        return nil, err
    }
    
    req.Header.Set("Authorization", "Bearer "+os.Getenv("SUPERMEMORY_API_KEY"))
    req.Header.Set("Content-Type", "application/json")
    
    client := &http.Client{}
    return client.Do(req)
}

func main() {
    // Get user
    resp, err := makeRequest("GET", "/v1/user", nil)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()
    
    // Add memory
    memoryData := map[string]interface{}{
        "type":    "note",
        "content": "This is a test note",
    }
    
    resp, err = makeRequest("POST", "/v1/add", memoryData)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()
}
```

### Key Validation

Always validate your API key before using it in production:

```javascript
async function validateApiKey(apiKey) {
  try {
    const response = await fetch('https://api.vevibe.com/v1/user', {
      headers: {
        'Authorization': `Bearer ${apiKey}`
      }
    });
    
    if (response.status === 401) {
      throw new Error('Invalid API key');
    }
    
    if (!response.ok) {
      throw new Error(`API validation failed: ${response.statusText}`);
    }
    
    const user = await response.json();
    console.log('API key validated for user:', user.email);
    return true;
  } catch (error) {
    console.error('API key validation failed:', error.message);
    return false;
  }
}

// Usage
const isValid = await validateApiKey(process.env.SUPERMEMORY_API_KEY);
if (!isValid) {
  process.exit(1);
}
```

## Error Handling

### Authentication Errors

Handle authentication-related errors gracefully:

```javascript
async function handleApiRequest(endpoint, options = {}) {
  try {
    const response = await fetch(`https://api.vevibe.com${endpoint}`, {
      ...options,
      headers: {
        'Authorization': `Bearer ${process.env.SUPERMEMORY_API_KEY}`,
        'Content-Type': 'application/json',
        ...options.headers,
      },
    });

    if (response.status === 401) {
      throw new Error('Authentication failed - check your API key');
    }

    if (response.status === 403) {
      throw new Error('Forbidden - insufficient permissions');
    }

    if (!response.ok) {
      const errorData = await response.json().catch(() => ({}));
      throw new Error(errorData.message || `HTTP ${response.status}: ${response.statusText}`);
    }

    return response.json();
  } catch (error) {
    console.error('API request failed:', error.message);
    throw error;
  }
}
```

### Common Error Responses

<Tabs>
  <Tab title="401 Unauthorized">
    ```json
    {
      "error": {
        "code": "UNAUTHORIZED",
        "message": "Invalid or missing API key"
      }
    }
    ```
    
    **Causes:**
    - Missing Authorization header
    - Invalid API key
    - Expired API key
    
    **Solutions:**
    - Verify your API key is correct
    - Check the Authorization header format
    - Generate a new API key if needed
  </Tab>
  
  <Tab title="403 Forbidden">
    ```json
    {
      "error": {
        "code": "FORBIDDEN", 
        "message": "Insufficient permissions for this resource"
      }
    }
    ```
    
    **Causes:**
    - Trying to access another user's resources
    - Account doesn't have required subscription level
    - API key doesn't have required permissions
    
    **Solutions:**
    - Verify you're accessing your own resources
    - Check your account subscription level
    - Ensure your API key has the required permissions
  </Tab>
  
  <Tab title="429 Rate Limited">
    ```json
    {
      "error": {
        "code": "RATE_LIMITED",
        "message": "Too many requests"
      }
    }
    ```
    
    **Causes:**
    - Exceeded rate limit for your plan
    - Too many concurrent requests
    
    **Solutions:**
    - Implement exponential backoff
    - Reduce request frequency
    - Consider upgrading your plan
  </Tab>
</Tabs>

## Multiple API Keys

For larger applications, you might need multiple API keys:

### Use Cases

- **Environment Separation**: Different keys for development, staging, production
- **Team Access**: Individual keys for team members
- **Service Separation**: Different keys for different services or applications
- **Rate Limit Distribution**: Multiple keys to increase effective rate limits

### Managing Multiple Keys

```javascript
class SuperMemoryClient {
  constructor(apiKeys = []) {
    this.apiKeys = Array.isArray(apiKeys) ? apiKeys : [apiKeys];
    this.currentKeyIndex = 0;
  }
  
  getCurrentApiKey() {
    return this.apiKeys[this.currentKeyIndex];
  }
  
  rotateApiKey() {
    this.currentKeyIndex = (this.currentKeyIndex + 1) % this.apiKeys.length;
  }
  
  async makeRequest(endpoint, options = {}) {
    let lastError;
    
    // Try each API key
    for (let attempt = 0; attempt < this.apiKeys.length; attempt++) {
      try {
        const response = await fetch(`https://api.vevibe.com${endpoint}`, {
          ...options,
          headers: {
            'Authorization': `Bearer ${this.getCurrentApiKey()}`,
            'Content-Type': 'application/json',
            ...options.headers,
          },
        });
        
        if (response.status === 429) {
          // Rate limited, try next key
          this.rotateApiKey();
          continue;
        }
        
        if (!response.ok) {
          throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }
        
        return response.json();
      } catch (error) {
        lastError = error;
        this.rotateApiKey();
      }
    }
    
    throw lastError;
  }
}

// Usage
const client = new SuperMemoryClient([
  process.env.SUPERMEMORY_API_KEY_1,
  process.env.SUPERMEMORY_API_KEY_2,
  process.env.SUPERMEMORY_API_KEY_3,
]);

const user = await client.makeRequest('/v1/user');
```

## API Key Lifecycle

### Key Rotation

Implement regular key rotation for enhanced security:

<Steps>
  <Step title="Generate New Key">
    Create a new API key in the SuperMemory dashboard
  </Step>
  <Step title="Update Applications">
    Gradually update applications to use the new key
  </Step>
  <Step title="Monitor Usage">
    Ensure all applications are using the new key
  </Step>
  <Step title="Revoke Old Key">
    Delete the old API key to complete the rotation
  </Step>
</Steps>

### Automated Rotation

```bash
#!/bin/bash
# rotate-api-key.sh - Automated API key rotation script

OLD_KEY=$SUPERMEMORY_API_KEY
NEW_KEY=$(generate-new-api-key.sh)  # Your key generation script

# Update environment
export SUPERMEMORY_API_KEY=$NEW_KEY

# Update configuration files
sed -i "s/$OLD_KEY/$NEW_KEY/g" /etc/myapp/config.yaml

# Restart services
systemctl restart myapp

# Verify new key works
if curl -f -H "Authorization: Bearer $NEW_KEY" https://api.vevibe.com/v1/user > /dev/null 2>&1; then
    echo "New API key validated successfully"
    # Revoke old key (implement your revocation logic here)
    revoke-api-key.sh $OLD_KEY
else
    echo "New API key validation failed, rolling back"
    export SUPERMEMORY_API_KEY=$OLD_KEY
    systemctl restart myapp
    exit 1
fi
```

## Testing Authentication

### Unit Tests

```javascript
// auth.test.js
const { describe, it, expect } = require('@jest/globals');

describe('API Authentication', () => {
  it('should include Authorization header', async () => {
    const mockFetch = jest.fn().mockResolvedValue({
      ok: true,
      json: () => Promise.resolve({ id: 123 })
    });
    
    global.fetch = mockFetch;
    
    await apiRequest('/v1/user');
    
    expect(mockFetch).toHaveBeenCalledWith(
      'https://api.vevibe.com/v1/user',
      expect.objectContaining({
        headers: expect.objectContaining({
          'Authorization': `Bearer ${process.env.SUPERMEMORY_API_KEY}`
        })
      })
    );
  });
  
  it('should handle 401 errors', async () => {
    const mockFetch = jest.fn().mockResolvedValue({
      ok: false,
      status: 401,
      statusText: 'Unauthorized'
    });
    
    global.fetch = mockFetch;
    
    await expect(apiRequest('/v1/user')).rejects.toThrow('Authentication failed');
  });
});
```
